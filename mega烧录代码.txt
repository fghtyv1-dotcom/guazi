#include <SimpleFOC.h>

#define IN1 44
#define IN2 45
#define IN3 46
#define EN  22

MagneticSensorI2C sensor = MagneticSensorI2C(AS5600_I2C);
BLDCMotor motor = BLDCMotor(7);
BLDCDriver3PWM driver = BLDCDriver3PWM(IN1, IN2, IN3);

Commander commander = Commander(Serial);

// 行程硬限制 -90 ~ 90 °
const float ANGLE_MAX = 90.0f;
const float ANGLE_MIN = -90.0f;
const float FULL_CIRCLE = 360.0f;
const float POS_ERR_THRESHOLD = 5.0f;
const unsigned long HOME_TIMEOUT = 2500;

bool is_homing = false;
unsigned long target_set_time = 0;
float last_target_deg = 0.0f;
float origin_angle = 0.0f;

// 角度归一到 [-180, 180]
float normalizeAngle(float deg)
{
  while (deg > 180.0f)  deg -= FULL_CIRCLE;
  while (deg < -180.0f) deg += FULL_CIRCLE;
  return deg;
}

// 获取当前相对角度
float getCurrentRelDeg()
{
  float rel_rad = motor.shaft_angle - origin_angle;
  float rel_deg = rel_rad * RAD_TO_DEG;
  return normalizeAngle(rel_deg);
}

// 指令处理 + 强制区间锁定
void doAngleCmd(char* cmd) {
  if(is_homing) return;
  float deg = 0.0f;
  commander.scalar(&deg, cmd);
  if (deg > ANGLE_MAX) deg = ANGLE_MAX;
  if (deg < ANGLE_MIN) deg = ANGLE_MIN;
  last_target_deg = deg;
  motor.target = origin_angle + deg * DEG_TO_RAD;
  target_set_time = millis();
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  Wire.begin();
  pinMode(EN, OUTPUT);
  digitalWrite(EN, HIGH);

  sensor.init();
  motor.linkSensor(&sensor);

  driver.voltage_power_supply = 12;
  driver.voltage_limit = 7.0;    //驱动上限拉高
  driver.init();
  motor.linkDriver(&driver);

  motor.controller = MotionControlType::angle;
  motor.voltage_limit = 6.2;     //工作电压加大，提速+增扭力
  motor.velocity_limit = 12;     //放开最大转速，解决走得慢

  motor.P_angle.P = 2.2;         //位置环增益加大，响应更快
  motor.PID_velocity.P = 0.09;
  motor.PID_velocity.I = 0.5;
  motor.PID_velocity.D = 0.0;
  motor.LPF_velocity.Tf = 0.02;
  motor.voltage_sensor_align = 0.9f; //校准电压提高，带载初始化顺畅

  motor.useMonitoring(Serial);
  motor.monitor_downsample = 100;
  motor.monitor_variables = _MON_TARGET | _MON_ANGLE | _MON_VEL | _MON_VOLT_Q;

  motor.init();
  motor.initFOC();
  delay(500);

  origin_angle = sensor.getAngle();
  last_target_deg = 0.0f;
  is_homing = true;
  motor.target = origin_angle;
  target_set_time = millis();
  Serial.println("上电开始自动回零至 0° ...");

  commander.add('T', doAngleCmd, "目标角度 ° [-90 ~ 90]");
  Serial.println("==============================");
  Serial.println("指令：T0 回零 | T90 正向最大 | T-90 反向最大");
  Serial.println("严格限制在 -90 ~ +90°");
  Serial.println("==============================");
}

void loop() {
  digitalWrite(EN, HIGH);
  float current_deg = getCurrentRelDeg();

  if(is_homing)
  {
    float pos_error = fabs(0.0f - current_deg);
    if(pos_error < POS_ERR_THRESHOLD)
    {
      is_homing = false;
      target_set_time = millis();
      Serial.println("上电回零完成，当前位置：0°");
    }
    motor.loopFOC();
    motor.move();
  }
  else
  {
    motor.loopFOC();
    motor.move();
  }

  commander.run();

  static unsigned long t = 0;
  if (millis() - t > 500)
  {
    t = millis();
    Serial.print("当前相对角度(°): ");
    Serial.print(current_deg, 1);
    Serial.print("  转速: ");
    Serial.print(motor.shaft_velocity, 4);
    if(is_homing) Serial.print("  状态: 正在回零");
    Serial.println();
  }
}
